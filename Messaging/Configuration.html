def initialize
  @publish_to       = "amqp://guest:guest@localhost:5672"
  @consume_from     = [publish_to]
  @prefetch         = 1
  @exchange_options = { :auto_delete => false, :durable => true }
  @queue_options    = exchange_options
  @reconnect_delay  = 5
  @logger           = Logger.new(STDOUT)
end
