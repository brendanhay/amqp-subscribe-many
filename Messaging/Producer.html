def publish(exchange, key, payload)
  ex = @exchanges[exchange] ||=
    declare_exchange(channel, exchange, EXCHANGE_TYPE)

  ex.publish(payload, {
    :exchange    => exchange,
    :routing_key => key
  })

  self
end
